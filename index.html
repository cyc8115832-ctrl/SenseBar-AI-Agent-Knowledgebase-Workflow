Python 腳本 - 用於抓取頻道列表並篩選關鍵字 
import subprocess
import json
import os
import sys

# 取得目標頻道 URL 或使用三師爸預設頻道
url = sys.argv[1] if len(sys.argv) > 1 else "https://www.youtube.com/@sensebar/videos"
yt_dlp_path = "./yt-dlp.exe"

# 強制 Windows 輸出環境為 UTF-8 以免中文亂碼
env = os.environ.copy()
env["PYTHONIOENCODING"] = "utf-8"

cmd = [
    yt_dlp_path,
    "--flat-playlist",
    "--print",
    '{"title": %(title)q, "url": %(url)q}',
    url
]

print(f"Fetching videos from {url}...")
result = subprocess.run(cmd, capture_output=True, env=env)
if result.returncode != 0:
    print("Error:", result.stderr.decode('utf-8', errors='replace'))
    sys.exit(1)

lines = result.stdout.decode('utf-8', errors='replace').strip().split("\n")
videos = []
for line in lines:
    if line.strip():
        try:
            videos.append(json.loads(line))
        except:
            continue

# 篩選核心 AI Agent 關鍵字
keywords = ["claude", "codex", "antigravity", "opencode", "agent"]
matched = []
for v in videos:
    title_lower = v['title'].lower()
    if any((kw in title_lower or (kw == "opencode" and "open code" in title_lower)) for kw in keywords):
        matched.append(v)

# 儲存篩選出的影片清單
with open("urls_list.txt", "w", encoding="utf-8") as f:
    for mv in matched:
        f.write(f"{mv['url']}\n")
        
print(f"成功儲存 {len(matched)} 部符合篩選條件的影片網址至 urls_list.txt")
Python 腳本 - 自動分流處理 CC 字幕清理與 Whisper 轉譯 
import os
import subprocess
import json
import re
import glob

workspace_dir = "."
yt_dlp_path = "./yt-dlp.exe"  # 本地 Windows 執行檔
urls_file = "urls_list.txt"

# 載入 Whisper (本地語音識別備案)
try:
    import whisper
    whisper_model = whisper.load_model("base")  # 可視需求調整為 small / medium
except ImportError:
    print("本機未安裝 Whisper。無 CC 字幕的影片將被跳過。請執行 'pip install openai-whisper torch' 啟用。")
    whisper_model = None

def clean_filename(filename):
    filename = re.sub(r'[\\/*?:"<>|]', "_", filename)
    filename = re.sub(r'_+', '_', filename)
    return filename.strip()

def parse_vtt(vtt_path):
    if not os.path.exists(vtt_path):
        return ""
    with open(vtt_path, "r", encoding="utf-8") as f:
        content = f.read()
    lines = content.split("\n")
    clean_lines = []
    for line in lines:
        line = line.strip()
        if not line or line.startswith("WEBVTT") or line.startswith("Kind:") or line.startswith("Language:") or "-->" in line:
            continue
        line = re.sub(r'<[^>]+>', '', line).strip() # 去除 HTML 標籤
        if not line:
            continue
        # 清理並去重 VTT 自動滾動字幕重複疊加的行
        if not clean_lines:
            clean_lines.append(line)
        else:
            last = clean_lines[-1]
            if line.startswith(last):
                clean_lines[-1] = line
            elif last.startswith(line):
                continue
            else:
                clean_lines.append(line)
    return "\n\n".join(clean_lines)

# 讀取待下載清單
if not os.path.exists(urls_file):
    print("找不到 urls_list.txt。請先執行 fetch_urls.py。")
    exit(0)
    
with open(urls_file, "r", encoding="utf-8") as f:
    urls = [line.strip() for line in f if line.strip()]

env = os.environ.copy()
env["PYTHONIOENCODING"] = "utf-8"

# 針對 Windows，動態從 Registry 載入當前使用者的系統 Path 避免 ffmpeg 找不到
import winreg
def get_env_paths():
    paths = []
    try:
        with winreg.OpenKey(winreg.HKEY_LOCAL_MACHINE, r"SYSTEM\CurrentControlSet\Control\Session Manager\Environment") as key:
            paths.append(winreg.QueryValueEx(key, "Path")[0])
    except: pass
    try:
        with winreg.OpenKey(winreg.HKEY_CURRENT_USER, r"Environment") as key:
            paths.append(winreg.QueryValueEx(key, "Path")[0])
    except: pass
    return ";".join(paths)

env["PATH"] = get_env_paths()

for idx, url in enumerate(urls, 1):
    print(f"\n[{idx}/{len(urls)}] 處理中: {url}")
    
    # 1. 抓取影片 Info JSON
    info_cmd = [yt_dlp_path, "--skip-download", "--dump-json", url]
    res = subprocess.run(info_cmd, capture_output=True, env=env)
    if res.returncode != 0:
        continue
    try:
        info_json = json.loads(res.stdout.decode('utf-8'))
        title = info_json.get("title")
        video_id = info_json.get("id")
        subtitles = info_json.get("subtitles", {})
        auto_subtitles = info_json.get("automatic_captions", {})
    except Exception as e:
        print(f"解析 JSON 失敗: {e}")
        continue

    print(f"影片標題: {title}")
    safe_title = clean_filename(title)
    md_path = f"{safe_title}.md"
    
    has_cc = len(subtitles) > 0 or len(auto_subtitles) > 0
    
    # 分支 A：有字幕 (CC) 檔案直接抓取並清理
    if has_cc:
        print("-> 偵測到影片中文字幕，啟動字幕下載...")
        temp_prefix = f"temp_sub_{video_id}"
        dl_cmd = [
            yt_dlp_path,
            "--skip-download",
            "--write-subs",
            "--write-auto-subs",
            "--sub-lang", "zh-Hant,zh-TW,zh,en",
            "--sub-format", "vtt",
            "-o", f"{temp_prefix}.%(ext)s",
            url
        ]
        subprocess.run(dl_cmd, capture_output=True)
        sub_files = glob.glob(f"{temp_prefix}.*")
        
        if sub_files:
            vtt_file = sub_files[0]
            clean_text = parse_vtt(vtt_file)
            
            with open(md_path, "w", encoding="utf-8") as f:
                f.write(f"# {title}\n\n- **影片連結**: {url}\n- **字幕來源**: YouTube CC 字幕清洗\n\n## 逐字稿\n\n{clean_text}\n")
            print(f"成功儲存: {md_path}")
            
            for sf in sub_files:
                os.remove(sf)
        else:
            print("無法成功下載 VTT 字幕，降級進入音訊轉譯流程...")
            has_cc = False
            
    # 分支 B：無字幕 (Non-CC)，下載音軌並呼叫 Whisper 本地轉譯
    if not has_cc:
        if not whisper_model:
            print("-> 無 CC 字幕且 Whisper 模組未安裝，跳過此影片。")
            continue
            
        print("-> 無字幕，啟動音訊下載進行 Whisper 本地語音辨識...")
        temp_audio = f"temp_audio_{video_id}.m4a"
        temp_mp3 = f"temp_audio_{video_id}.mp3"
        
        # 呼叫 node 解決 YouTube 阻擋 403 Forbidden 限速
        dl_cmd = [
            yt_dlp_path,
            "--js-runtimes", "node",
            "-f", "140",
            "-o", temp_audio,
            url
        ]
        dl_res = subprocess.run(dl_cmd, capture_output=True, env=env)
        
        if dl_res.returncode == 0 and os.path.exists(temp_audio):
            # 呼叫 FFmpeg 轉檔高品質 MP3
            ffmpeg_cmd = ["ffmpeg", "-y", "-i", temp_audio, "-acodec", "libmp3lame", "-aq", "2", temp_mp3]
            subprocess.run(ffmpeg_cmd, capture_output=True, env=env)
            os.remove(temp_audio)
            
            if os.path.exists(temp_mp3):
                print(f"正在以 Whisper base 模型辨識音訊: {temp_mp3}...")
                result = whisper_model.transcribe(temp_mp3, language="zh")
                transcribed_text = result.get("text", "")
                
                # 自動斷句分段
                formatted_text = "\n\n".join(re.split(r'(?<=[。！？])\s*', transcribed_text))
                
                with open(md_path, "w", encoding="utf-8") as f:
                    f.write(f"# {title}\n\n- **影片連結**: {url}\n- **字幕來源**: Whisper 本地語音辨識\n\n## 逐字稿\n\n{formatted_text}\n")
                print(f"轉譯完成！成功儲存: {md_path}")
                os.remove(temp_mp3)
            else:
                print("FFmpeg 轉檔音訊失敗。")
        else:
            print("音軌下載失敗。")
            
print("所有工作流程已執行完畢！")

# YouTube 影音知識庫建構器技能 (YouTube Knowledgebase Builder Skill)

此技能供 AI Agent 在需要從 YouTube 下載影片、音訊、提取與清理 CC 字幕、或針對無字幕影片進行 Whisper 本地轉錄時調用。

## ⚙️ 系統依賴要求
- **yt-dlp**：本機需具備 `yt-dlp` 執行檔（若在 Windows 系統，預設路徑為專案根目錄的 `./yt-dlp.exe`）。
- **Node.js**：簽章解密挑戰需要，執行 `yt-dlp` 時請務必帶上 `--js-runtimes node`。
- **FFmpeg**：音訊轉檔與合併所必需，請確保已加到系統 `PATH`。
- **Whisper (Python)**：無字幕影片轉錄需要，安裝指令：`pip install openai-whisper torch`。

## 🛠️ 常見指令與工作流

### 1. 抓取頻道影片清單並過濾關鍵字
呼叫 fetch_urls.py 腳本，從目標 YouTube 頻道中獲取所有影片，並過濾出目標關鍵字網址儲存至 `urls_list.txt`。

### 2. 獲取並清理逐字稿 (自動分流處理)
呼叫 build_knowledge_base.py 腳本，對 `urls_list.txt` 中的網址進行分析：
- **分支 A（有 CC）**：下載 VTT，並使用去重演算法清理重複滾動行。
- **分支 B（無 CC）**：下載 M4A，呼叫 FFmpeg 轉為 MP3，並使用本地 Whisper 轉錄。

---

## 💻 核心技術指令參考

### 下載字幕 (CC) 且不下載影片
```powershell
./yt-dlp.exe --skip-download --write-subs --write-auto-subs --sub-lang "zh-Hant,zh-TW,zh,en" --sub-format vtt -o "temp_sub.%(ext)s" "URL"
```

### 破解 403 Forbidden 限速下載 M4A 音訊
```powershell
./yt-dlp.exe --js-runtimes node -f 140 -o "temp_audio.m4a" "URL"
```

### 使用 FFmpeg 轉檔音訊為最高品質 MP3
```powershell
ffmpeg -y -i "temp_audio.m4a" -acodec libmp3lame -aq 0 "output_audio.mp3"
```
本機開發環境依賴設置指引 
🛠️ 步驟 1：安裝 Python 3.12 核心環境

在您的本機系統中安裝 Python。您可透過 PowerShell 執行以下指令自動安裝：
winget install --id Python.Python.3.12 --source winget --silent --accept-source-agreements --accept-package-agreements


⚙️ 步驟 2：下載 yt-dlp.exe

至 yt-dlp 官方 GitHub Release 頁面，拉至下方「Assets」區段，下載 yt-dlp.exe 並將其擺放於專案根目錄中。


🎙️ 步驟 3：安裝 FFmpeg 音訊處理器

在管理員權限的 PowerShell 中執行以下指令：
winget install Gyan.FFmpeg


🧠 步驟 4：安裝 Python 機器學習依賴庫（Whisper 轉譯）

在命令列中安裝 openai-whisper 語音轉文字套件與 PyTorch 硬體加速庫：
pip install openai-whisper torch


🚀 步驟 5：執行工作流

1. 將待下載的影片 URL 輸入 urls_list.txt（每行一個）。
2. 在終端機執行：
python build_knowledge_base.py
3. 程式將自動分析，完成後會在目錄下產出與影片標題同名的 .md 逐字稿檔案，可直接拖曳匯入 Obsidian 或同步至 GitHub。
